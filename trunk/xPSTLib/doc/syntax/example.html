Main .xpst file example

include "OffPath.xpst";

options
{
  tutorname="CAPE";
}

sequence
{
  (
    #---Problem-creation
    (Click-New or Error-Wrong-New) then
    (
      (
        Text-ProblemName and
        Text-Attempts and
        CheckBox-Start
      ) until
      Click-Update
    ) then
    ...
  ) and
  Error-Offpath;
}

mappings
{
  [priority=2] ProblemNew:click => Click-New;
  [priority=2] ProblemName => Text-ProblemName;
  [priority=2] ProblemEditControls.childNodes-3:click => Click-Update;
  ...
}

feedback
{
  Text-Attempts
  {
    answer: "2";
    Hint: "Set the number of attempts for this problem. This means that the students will have two opportunities to try out this problem and enter the correct answer.";
    Hint: "Set the <strong>Attempts</strong> field for this problem to '{answer}'.";
  }

  Text-Assess-Title
  {
    answer: RegEx(".+");
    Hint: "Provide a title for the problem. This title will appear at the top of the whole problem set displayed to the student.";
    Hint: "Type 'Add and Subtract Integers' for the <strong>Title</strong>.";
  }
}

Off-path example

mappings
{
  [priority=0]
  switch
  {
    AssessmentLink.childNodes-3:click="clicked on the Assessment view link",
    AssessmentButton.childNodes-0:click="clicked on the Assessment view link",

    FeedbackLink.childNodes-3:click="clicked on the Feedback view link",
    FeedbackButton.childNodes-0:click="clicked on the Feedback view link",
    ...
  } => Error-Offpath;
}

feedback
{
  Error-Offpath
  {
    answer: "foo";
    JIT: "<block/>You {v}. That's not something you need to do right now. Request a hint if you'd like to know what to do next.";
  }
}
